171  private int interpColor(int colors[], float unit) {
172  if (unit <= 0) {
173  return colors[0];
174  }
175  if (unit >= 1) {
176  return colors[colors.length - 1];
177  }
178 
179  float p = unit * (colors.length - 1);
180  int i = (int) p;
181  p -= i;
182 
183  // now p is just the fractional part [0...1) and i is the index
184  int c0 = colors[i];
185  int c1 = colors[i + 1];
186  int a = ave(Color.alpha(c0), Color.alpha(c1), p);
187  int r = ave(Color.red(c0), Color.red(c1), p);
188  int g = ave(Color.green(c0), Color.green(c1), p);
189  int b = ave(Color.blue(c0), Color.blue(c1), p);
190 
191  return Color.argb(a, r, g, b);
192  }

194  private int rotateColor(int color, float rad) {
195  float deg = rad * 180 / 3.1415927f;
196  int r = Color.red(color);
197  int g = Color.green(color);
198  int b = Color.blue(color);

200  ColorMatrix cm = new ColorMatrix();
201  ColorMatrix tmp = new ColorMatrix();
202 
203  cm.setRGB2YUV();
204  tmp.setRotate(0, deg);
205  cm.postConcat(tmp);
206  tmp.setYUV2RGB();
207  cm.postConcat(tmp);
208 
209  final float[] a = cm.getArray();
210 
211  int ir = floatToByte(a[0] * r + a[1] * g + a[2] * b);
212  int ig = floatToByte(a[5] * r + a[6] * g + a[7] * b);
213  int ib = floatToByte(a[10] * r + a[11] * g + a[12] * b);
214 
215  return Color.argb(Color.alpha(color), pinToByte(ir),
216  pinToByte(ig), pinToByte(ib));

221  @Override
222  public boolean onTouchEvent(MotionEvent event) {
223  float x = event.getX() - CENTER_X;
224  float y = event.getY() - CENTER_Y;
225  boolean inCenter = java.lang.Math.sqrt(x * x + y * y) <= CENTER_RADIUS;
226 
227  switch (event.getAction()) {
228  case MotionEvent.ACTION_DOWN:
229  mTrackingCenter = inCenter;
230  if (inCenter) {
231  mHighlightCenter = true;
232  invalidate();
233  break;
234  }
235  case MotionEvent.ACTION_MOVE:
236  if (mTrackingCenter) {
237  if (mHighlightCenter != inCenter) {
238  mHighlightCenter = inCenter;
239  invalidate();
240  }
241  } else {
242  float angle = (float) java.lang.Math.atan2(y, x);
243  // need to turn angle [-PI ... PI] into unit [0....1]
244  float unit = angle / (2 * PI);
245  if (unit < 0) {
246  unit += 1;
247  }
248  mCenterPaint.setColor(interpColor(mColors, unit));
249  invalidate();
250  }
251  break;
252  case MotionEvent.ACTION_UP:
253  if (mTrackingCenter) {
254  if (inCenter) {
255  if (mListener != null) {
256  mListener.colorChanged(mCenterPaint.getColor());
257  }
258  if (mContextListener != null) {
259  mContextListener.colorChanged(mCenterPaint.getColor());
260  }
261  }
262  mTrackingCenter = false; // so we draw w/o halo
263  invalidate();
264  }
265  break;
266  }
